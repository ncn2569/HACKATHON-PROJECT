LOADER_MAP = {
    ".txt":  lambda p: TextLoader(p, encoding="utf-8"),
    ".pdf":  lambda p: PyPDFLoader(p),
    ".docx": lambda p: Docx2txtLoader(p),
}

def load_all_documents(data_dir: str):
    all_docs = []
    data_path = Path(data_dir)

    if not data_path.exists():
        raise ValueError(f"❌ Thư mục '{data_dir}' không tồn tại")

    for file_path in data_path.rglob("*"):
        suffix = file_path.suffix.lower()
        if suffix not in LOADER_MAP:
            continue

        try:
            loader = LOADER_MAP[suffix](str(file_path))
            docs = loader.load()
            for doc in docs:
                doc.metadata["source_file"] = file_path.name
            all_docs.extend(docs)
            print(f"✅ [{suffix}] {file_path.name} → {len(docs)} trang/đoạn")
        except Exception as e:
            print(f"⚠️  Bỏ qua {file_path.name}: {e}")

    return all_docs

def format_docs(docs):
    return "\n\n".join(doc.page_content for doc in docs)

rag_chain = (
    {
        "context": itemgetter("question") | retriever | format_docs,
        "chat_history": itemgetter("chat_history"),
        "question": itemgetter("question"),
        "style": itemgetter("style")
    }
    | prompt
    | llm
)

# MEMORY STORE

MAX_HISTORY = 3

session_store: dict[str, deque] = {}

def get_chat_history(session_id: str) -> str:
    if session_id not in session_store or not session_store[session_id]:
        return "Chưa có lịch sử trò chuyện."

    lines = []
    for msg in session_store[session_id]:
        lines.append(f"- Học sinh: {msg['user']}\n- AI: {msg['ai']}")
    return "\n".join(lines)

def update_chat_history(session_id: str, user_msg: str, ai_response: str):
    if session_id not in session_store:
        session_store[session_id] = deque(maxlen=MAX_HISTORY)
    session_store[session_id].append({"user": user_msg, "ai": ai_response})

# =========================
# CLEAN JSON
# =========================
def clean(text):
    if not isinstance(text, str):
        text = str(text)

    text = re.sub(r"```json|```", "", text).strip()

    match = re.search(r"\{.*\}", text, re.DOTALL)
    if not match:
        return None

    try:
        return json.loads(match.group())
    except Exception:
        return None

# =========================
def fallback():
    return {
        "action": "ASK_BACK",
        "value": "",
        "text": "Bạn đã thử cách nào để giải bài này chưa?"
    }

STYLE_MAP = {
    "Tiêu chuẩn": "Giọng điệu chuẩn mực, rõ ràng, mang tính sư phạm, thân thiện và lịch sự.",

@app.post("/chat")
def chat(q: Query):
    try:
        # 1. Kiểm tra trạng thái bế tắc bằng từ khóa
        detailed_style = STYLE_MAP.get(q.style, STYLE_MAP["Tiêu chuẩn"])
        stuck_keywords = ["không biết", "chịu", "giải hộ", "đáp án là gì", "không hiểu", "khó quá"]
        is_stuck = any(k in q.question.lower() for k in stuck_keywords)
        
        # Kiểm tra xem session này đã có lịch sử trò chuyện chưa 
        # (Nếu len > 0 tức là đã có ít nhất 1 lượt AI phản hồi/hỏi ngược trước đó)
        has_history = False
        if q.session_id in session_store and len(session_store[q.session_id]) > 0:
            has_history = True

        # 2. Bơm lệnh ép buộc (Injection) nếu đủ điều kiện
        actual_question = q.question
        if is_stuck and has_history:
            # Ép model phải dùng EXPLAIN và nhả lời giải
            actual_question += "\n\n[LỆNH TỪ HỆ THỐNG: Học sinh đã bế tắc và đã qua ít nhất 1 lượt gợi ý. BẮT BUỘC sử dụng action EXPLAIN và hiển thị ĐÁP ÁN, LỜI GIẢI CHI TIẾT từ context. KHÔNG hỏi ngược nữa.]"

        # 3. Lấy chuỗi lịch sử trò chuyện
        history_str = get_chat_history(q.session_id)

        # 4. Gọi RAG chain với câu hỏi đã được xử lý
        result = rag_chain.invoke({
            "question": actual_question, # Dùng câu hỏi đã được bơm lệnh (nếu có)
            "chat_history": history_str,
            "style": detailed_style
        })
        raw = result.content

        # Xử lý trường hợp content là list
        if isinstance(raw, list):
            raw = "".join([
                part.get("text", "") if isinstance(part, dict) else str(part)
                for part in raw
            ])

        parsed = clean(raw)
        used_fallback = parsed is None

        if not parsed:
            parsed = fallback()

        parsed.setdefault("action", "")
        parsed.setdefault("value", "")
        parsed.setdefault("text", "")

        # 5. Lưu lịch sử (QUAN TRỌNG: Chỉ lưu câu hỏi gốc của user, không lưu phần lệnh bơm thêm)
        if not used_fallback:
            update_chat_history(q.session_id, q.question, parsed["text"])

        return parsed

    except Exception as e:
        print("ERROR:", e)
        return {
            "action": "ERROR",
            "value": "",
            "text": "Server lỗi"
        }

Học tập hiệu quả không phải là việc nhớ đáp án mà là hiểu cách giải quyết vấn đề.

Hiện tượng "brainrot" xảy ra khi người học phụ thuộc vào AI để lấy đáp án mà không suy nghĩ.

Một hệ thống học tập tốt cần:
- Khuyến khích tư duy
- Hỏi ngược lại người học
- Cá nhân hóa nội dung

--------------------------------

Khi học sinh hỏi:
"Đáp án bài này là gì?"

Không nên trả lời trực tiếp.
Thay vào đó nên hỏi:
- Em đã thử cách nào rồi?
- Em nghĩ bước đầu tiên là gì?
- Bài này liên quan kiến thức nào?

--------------------------------

Các phương pháp học hiệu quả:

1. Active Recall:
Tự nhớ lại kiến thức thay vì đọc lại

2. Spaced Repetition:
Ôn tập theo khoảng thời gian

3. Feynman Technique:
Giải thích lại bằng ngôn ngữ đơn giản

--------------------------------

Ví dụ:

Toán:
2 + 2 = ?
→ hỏi: Em có thể cộng 2 với 2 bằng cách nào?

Lập trình:
"Code này lỗi ở đâu?"
→ hỏi: Em đã debug thử chưa? Lỗi xuất hiện ở dòng nào?

--------------------------------

Adaptive Quiz:

Nếu học sinh làm đúng:
→ tăng độ khó

Nếu sai:
→ giảm độ khó + giải thích

--------------------------------

Phân tích học sinh:

- Nếu hỏi liên tục đáp án → thiếu tư duy
- Nếu hỏi cách làm → đang hiểu

--------------------------------

Gợi ý lộ trình:

Beginner:
- Học lý thuyết cơ bản
- Làm bài đơn giản

Intermediate:
- Làm bài thực tế
- Debug lỗi

Advanced:
- Làm project
- Tối ưu code

--------------------------------

Vai trò của AI:

- Không thay thế tư duy
- Chỉ hướng dẫn
- Đặt câu hỏi đúng

--------------------------------

Mục tiêu cuối cùng:

Giúp người học:
- Tư duy độc lập
- Hiểu bản chất
- Không phụ thuộc AI
